Informacion tecnica

Esta operacion utiliza una transaccion SQL explicita para registrar la venta, insertar sus detalles y descontar inventario. Si hay stock insuficiente o un error de base de datos, se ejecuta ROLLBACK.

BEGINCOMMITROLLBACK
BEGIN;

SELECT id_producto, nombre, stock, precio_unitario
FROM producto
WHERE id_producto = :id_producto
FOR UPDATE;

-- Si cantidad solicitada mayor a stock:
ROLLBACK;

INSERT INTO venta (fecha, total, estado, id_cliente, id_empleado)
VALUES (:fecha, :total, :estado, :id_cliente, :id_empleado);

UPDATE producto
SET stock = stock - :cantidad
WHERE id_producto = :id_producto;

INSERT INTO detalle_venta (id_venta, id_producto, cantidad, precio_unitario, subtotal)
VALUES (:id_venta, :id_producto, :cantidad, :precio_unitario, :subtotal);

COMMIT;
